CREATE DATABASE IF NOT EXISTS clinica;
USE clinica;



CREATE TABLE Pacientes (
    Codigo_Paciente INT PRIMARY KEY,
    Nome VARCHAR(100),
    Idade INT,
    Cargo VARCHAR(50),
    Historico_medico TEXT,
    Data_nascimento DATE
);

CREATE TABLE Profissionais_enfermaria (
    Codigo_profissional INT PRIMARY KEY,
    Nome VARCHAR(100),
    Funcao VARCHAR(50),
    Especialidade VARCHAR(100)
);



CREATE TABLE Atendimento (
    Codigo_atendimento INT PRIMARY KEY,
    Codigo_profissional INT,
    Codigo_Paciente INT,
    Data DATE,
    Hora TIME,
    Status VARCHAR(50),
    FOREIGN KEY (Codigo_profissional) REFERENCES Profissionais_enfermaria(Codigo_profissional),
    FOREIGN KEY (Codigo_Paciente) REFERENCES Pacientes(Codigo_Paciente)
);



CREATE TABLE Diagnostico (
    Codigo_Diagnostico INT PRIMARY KEY,
    idPacientes INT,
    idProfissionais_enfermaria INT,
    Descricao TEXT,
    Suspeita_identificada VARCHAR(100),
    Data_avaliacao DATE,
    FOREIGN KEY (idPacientes) REFERENCES Pacientes(Codigo_Paciente),
    FOREIGN KEY (idProfissionais_enfermaria) REFERENCES Profissionais_enfermaria(Codigo_profissional)
);



CREATE TABLE Receita_Tratamentos (
    Codigo_receita INT NOT NULL,
    Codigo_tratamento INT NOT NULL,
    Dosagem VARCHAR(50),
    Nome_formula VARCHAR(100),
    Data_prescricao DATE,
    Duracao VARCHAR(50),
    Status VARCHAR(50),
    Nome_tratamento VARCHAR(100),
    PRIMARY KEY (Codigo_receita, Codigo_tratamento),
    -- Surrogate unique indexes so FKs in Resultado and Cria can target
    -- each column individually (required by most SQL engines for FK targets).
    UNIQUE KEY uq_receita     (Codigo_receita),
    UNIQUE KEY uq_tratamento  (Codigo_tratamento)
);

CREATE TABLE Resultado (
    Codigo_tratamento INT,
    Codigo_Diagnostico INT,
    Caso_paciente VARCHAR(500),
    PRIMARY KEY (Codigo_tratamento, Codigo_Diagnostico),
    FOREIGN KEY (Codigo_tratamento)  REFERENCES Receita_Tratamentos(Codigo_tratamento),
    FOREIGN KEY (Codigo_Diagnostico) REFERENCES Diagnostico(Codigo_Diagnostico)
);

CREATE TABLE Cria (
    Codigo_receita INT,
    Codigo_profissional INT,
    PRIMARY KEY (Codigo_receita, Codigo_profissional),
    FOREIGN KEY (Codigo_receita)      REFERENCES Receita_Tratamentos(Codigo_receita),
    FOREIGN KEY (Codigo_profissional) REFERENCES Profissionais_enfermaria(Codigo_profissional)
);

CREATE TABLE Ingredientes (
    Codigo_ingrediente INT PRIMARY KEY,
    Nome VARCHAR(100),
    Tipo VARCHAR(50),
    Quantidade_disponivel INT,
    Origem VARCHAR(100),
    Nivel_toxicidade VARCHAR(50)
);



CREATE TABLE Criacao (
    Codigo_receita INT,
    Codigo_ingrediente INT,
    Quantidade_ingrediente INT,
    PRIMARY KEY (Codigo_receita, Codigo_ingrediente),
    FOREIGN KEY (Codigo_receita)     REFERENCES Receita_Tratamentos(Codigo_receita),
    FOREIGN KEY (Codigo_ingrediente) REFERENCES Ingredientes(Codigo_ingrediente)
);



CREATE TABLE Estatisticas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    numero_pacientes INT DEFAULT 0,
    numero_profissionais INT DEFAULT 0,
    numero_pacientes_atendimento INT DEFAULT 0
);


DELIMITER $$

CREATE PROCEDURE IF NOT EXISTS pr_adicionar_paciente(
    p_codigo INT,
    p_nome VARCHAR(100),
    p_idade INT,
    p_cargo VARCHAR(50),
    p_historico TEXT,
    p_data_nasc DATE
) 
BEGIN
    INSERT INTO Pacientes (Codigo_Paciente, Nome, Idade, Cargo, Historico_medico, Data_nascimento)
    VALUES (p_codigo, p_nome, p_idade, p_cargo, p_historico, p_data_nasc);
END$$
DELIMITER ;

DELIMITER$$
CREATE PROCEDURE IF NOT EXISTS pr_adicionar_profissional(
    p_codigo INT,
    p_nome VARCHAR(100),
    p_funcao VARCHAR(50),
    p_especialidade VARCHAR(100)
) 
BEGIN
    INSERT INTO Profissionais_enfermaria (Codigo_profissional, Nome, Funcao, Especialidade)
    VALUES (p_codigo, p_nome, p_funcao, p_especialidade);
END$$

DELIMITER ;


DELIMITER $$

CREATE PROCEDURE IF NOT EXISTS sp_get_receitas_com_ingredientes()
BEGIN
    SELECT
        rt.Codigo_receita,
        rt.Nome_formula,
        rt.Dosagem,
        rt.Data_prescricao,
        i.Nome          AS Ingrediente_Nome,
        i.Tipo,
        i.Nivel_toxicidade,
        c.Quantidade_ingrediente
    FROM Receita_Tratamentos rt
    INNER JOIN Criacao c
        ON c.Codigo_receita = rt.Codigo_receita
    INNER JOIN Ingredientes i
        ON i.Codigo_ingrediente = c.Codigo_ingrediente
    ORDER BY rt.Codigo_receita, i.Nome;
END$$

CREATE FUNCTION IF NOT EXISTS fc_get_pacientes_com_tratamento()
BEGIN
    SELECT
        p.Codigo_Paciente,
        p.Nome          AS Paciente_Nome,
        p.Idade,
        p.Cargo,
        d.Suspeita_identificada,
        d.Data_avaliacao,
        rt.Nome_tratamento,
        rt.Dosagem,
        rt.Status       AS Status_Tratamento,
        rt.Duracao
    FROM Pacientes p
    LEFT JOIN Diagnostico d
        ON d.idPacientes = p.Codigo_Paciente
    LEFT JOIN Resultado r
        ON r.Codigo_Diagnostico = d.Codigo_Diagnostico
    LEFT JOIN Receita_Tratamentos rt
        ON rt.Codigo_tratamento = r.Codigo_tratamento
    ORDER BY p.Codigo_Paciente;
END$$

DELIMITER ;


INSERT INTO Estatisticas (id, numero_pacientes, numero_profissionais, numero_pacientes_atendimento) 
VALUES (1, 0, 0, 0)
ON DUPLICATE KEY UPDATE id=id;

CREATE TRIGGER trg_after_insert_paciente
AFTER INSERT ON Pacientes
FOR EACH ROW
BEGIN
    UPDATE Estatisticas 
    SET numero_pacientes = numero_pacientes + 1 
    WHERE id = 1;
END$$

Create View ViewClinica AS
SELECT
            p.Nome              AS Paciente,
            p.Idade,
            pr.Nome             AS Profissional,
            pr.Especialidade,
            a.Data              AS Data_Atendimento,
            a.Status            AS Status_Atendimento,
            d.Suspeita_identificada AS Suspeita,
            rt.Nome_tratamento  AS Tratamento,
            rt.Dosagem,
            rt.Duracao,
            rt.Status           AS Status_Tratamento
        FROM Atendimento a
        INNER JOIN Pacientes p
            ON a.Codigo_Paciente = p.Codigo_Paciente
        INNER JOIN Profissionais_enfermaria pr
            ON a.Codigo_profissional = pr.Codigo_profissional
        LEFT JOIN Diagnostico d
            ON d.idPacientes = p.Codigo_Paciente
           AND d.idProfissionais_enfermaria = pr.Codigo_profissional
        LEFT JOIN Resultado r
            ON r.Codigo_Diagnostico = d.Codigo_Diagnostico
        LEFT JOIN Receita_Tratamentos rt
            ON rt.Codigo_tratamento = r.Codigo_tratamento
        ORDER BY a.Data DESC;

DELIMITER $$

CREATE PROCEDURE IF NOT EXISTS sp_get_ingredientes_estoque_baixo(
    p_limite INT
)
BEGIN
    SELECT
        Codigo_ingrediente,
        Nome,
        Tipo,
        Quantidade_disponivel,
        Origem,
        Nivel_toxicidade
    FROM Ingredientes
    WHERE Quantidade_disponivel <= p_limite
    ORDER BY Quantidade_disponivel ASC;
END$$

CREATE PROCEDURE IF NOT EXISTS sp_get_historico_completo_paciente(
    p_codigo INT
)
BEGIN
    SELECT
        p.Nome                  AS Paciente,
        a.Data                  AS Data_Atendimento,
        a.Hora,
        a.Status                AS Status_Atendimento,
        pr.Nome                 AS Profissional,
        pr.Especialidade,
        d.Descricao             AS Descricao_Diagnostico,
        d.Suspeita_identificada,
        d.Data_avaliacao,
        rt.Nome_tratamento      AS Tratamento,
        rt.Dosagem,
        rt.Duracao,
        rt.Status               AS Status_Tratamento
    FROM Pacientes p
    LEFT JOIN Atendimento a
        ON a.Codigo_Paciente = p.Codigo_Paciente
    LEFT JOIN Profissionais_enfermaria pr
        ON a.Codigo_profissional = pr.Codigo_profissional
    LEFT JOIN Diagnostico d
        ON d.idPacientes = p.Codigo_Paciente
       AND d.idProfissionais_enfermaria = pr.Codigo_profissional
    LEFT JOIN Resultado r
        ON r.Codigo_Diagnostico = d.Codigo_Diagnostico
    LEFT JOIN Receita_Tratamentos rt
        ON rt.Codigo_tratamento = r.Codigo_tratamento
    WHERE p.Codigo_Paciente = p_codigo
    ORDER BY a.Data DESC, a.Hora DESC;
END$$

DELIMITER ;

DELIMITER $$

CREATE FUNCTION IF NOT EXISTS fc_contar_ingredientes_em_estoque()
RETURNS INT
READS SQL DATA
BEGIN
    DECLARE total INT;
    SELECT COALESCE(SUM(Quantidade_disponivel), 0) INTO total FROM Ingredientes;
    RETURN total;
END$$

CREATE FUNCTION IF NOT EXISTS fc_contar_receitas()
RETURNS INT
READS SQL DATA
BEGIN
    DECLARE total INT;
    SELECT COUNT(*) INTO total FROM Receita_Tratamentos;
    RETURN total;
END$$

DELIMITER ;


DELIMITER $$

CREATE FUNCTION IF NOT EXISTS fc_contar_uso_ingrediente(p_codigo_ingrediente INT)
RETURNS INT
READS SQL DATA
BEGIN
    DECLARE total INT;
    SELECT COUNT(*) INTO total
    FROM Criacao
    WHERE Codigo_ingrediente = p_codigo_ingrediente;
    RETURN total;
END$$

DELIMITER ;
